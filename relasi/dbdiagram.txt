Table Pengguna {
  id_pengguna INT [pk, increment]
  username VARCHAR [unique, not null]
  email VARCHAR [unique, not null]
  password_hash VARCHAR [not null]
  nama_lengkap VARCHAR [not null]
}

Table Kucing {
  id_kucing INT [pk, increment]
  nama VARCHAR [not null]
  jenis VARCHAR [not null]
  tipe_kucing ENUM('pengguna', 'adopsi') [not null]
  usia INT [not null]
  berat FLOAT [not null]
  gender ENUM('jantan', 'betina') [not null]
  kesehatan VARCHAR
  lokasi_penampungan VARCHAR
  deskripsi TEXT
  kontak_penampungan VARCHAR
  status_adopsi BOOLEAN [default: 0]
  foto_kucing VARCHAR
  id_pengguna INT [ref: > Pengguna.id_pengguna]
}

Table JadwalPemberianMakan {
  id_detail INT [pk, increment]
  id_kucing INT [ref: > Kucing.id_kucing]
  berapa_kali_makan INT [not null]
  waktu_makan TIME [not null]
  kebutuhan_kalori INT
}

Table ArtikelKucing {
  id_artikel INT [pk, increment]
  judul VARCHAR [not null]
  konten TEXT [not null]
  tanggal_publikasi DATE [not null]
  penulis VARCHAR [not null]
  foto_artikel VARCHAR
}

Table Diskusi {
  id_topik INT [pk, increment]
  konten_topik TEXT [not null]
  waktu_post DATETIME [not null]
  id_pengguna INT [ref: > Pengguna.id_pengguna]
}

Table Balasan {
  id_balasan INT [pk, increment]
  id_topik INT [ref: > Diskusi.id_topik]
  id_parent_balasan INT [ref: > Balasan.id_balasan]
  konten_balasan TEXT [not null]
  waktu_balasan DATETIME [not null]
  id_pengguna INT [ref: > Pengguna.id_pengguna]
}

Table RiwayatPemberianMakan {
  id_riwayat INT [pk, increment]
  id_kucing INT [ref: > Kucing.id_kucing]
  waktu_makan DATETIME [not null]
  jumlah_pemberian INT [not null]
}
